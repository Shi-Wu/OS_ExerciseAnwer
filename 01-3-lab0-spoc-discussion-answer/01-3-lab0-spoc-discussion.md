#include <u.h>
int ret;
out(port, val)
{
  asm(LL,8);   // load register a with port
  asm(LBL,16); // load register b with val
  asm(BOUT);   // output byte to console
}

int write(int f, char *s, int n)
{
  int i;
  ret = 1;
  i=n;
  while (i--)
    out(f, *s++);
  return i;
}

   - ### funcall中的堆栈有多大？是内核态堆栈还是用户态堆栈

	- 放在栈底的上面，利用sp
